Accordion

Javascript
var acc = document.getElementByClassName("accordion");
var item;

for(item = 0; item < acc.length; item++){
 acc[item].onclick = function(){
  this.classList.toggle("active");
  this.nextElementSibling.classList.toggle("show");
 }
}
